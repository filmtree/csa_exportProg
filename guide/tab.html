	<script>
		tabList();
		function tabList() {
		// 설계안수립 이너탭
		const tabItem = document.querySelectorAll('.tab-base > .inner-tab > li > a');
		const tabCont = document.querySelectorAll('.tab-base > .inner-contents');
		const tabInnerItem = document.querySelectorAll('.tab-innerdouble .double-tab > li > a');
		const tabInnerCont = document.querySelectorAll('.tab-innerdouble .double-contents');
		tabItem.forEach((item, index) => {
			item.addEventListener("click", function () {
				this.closest('.inner-tab').querySelectorAll('li > a').forEach(e => {
					e.classList.remove('active');
					event.preventDefault();
				})
				this.classList.add('active');
				this.closest('.wrap-inner').querySelectorAll('.inner-contents').forEach(e => {
					e.style.display = "none";
				});
				tabCont[index].style.display = "block";
			});
		});
	
		tabInnerItem.forEach((tabitem, index1) => {
			tabitem.addEventListener("click", function () {     
				this.closest('.tab-innerdouble').querySelectorAll('li > a').forEach(e => {
					e.classList.remove('active');
					event.preventDefault();
				})
				this.classList.add('active');
				this.closest('.tab-innerdouble').querySelectorAll('.double-contents').forEach(e => {
					e.style.display = "none";
				});
				tabInnerCont[index1].style.display = "block";
			});
		});
	}
	</script>
